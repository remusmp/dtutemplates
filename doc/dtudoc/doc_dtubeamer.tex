\section{User Guide}
\frame{
	\frametitle{Installation}
	The zip file contents complies with the TDS structure.
	
	For Unix and Mac systems:
	\begin{enumerate}
		\item Unzip the bundle to \texttt{\textasciitilde/texmf/}.
	\end{enumerate}
	
	For Windows systems:
	\begin{enumerate}
		\item Unzip the bundle to any location.
		\item Open Miktex settings.
		\item Add location to root folders.
		\item Refresh FNDB.
	\end{enumerate}
}
